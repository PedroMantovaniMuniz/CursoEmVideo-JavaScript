Funções são blocos de código executados após uma clamada ou em decorrência de eventos. Elas podem receber parâmetros e retornar valores.

function numPar(n) {
    if (n % 2 === 0) {
        return 'par';
    } else {
        return 'ímpar';
    }
}
let res = numPar(5); // Chamada da função numPar() com o parâmetro 5. A variável terá o valor 'ímpar'.
